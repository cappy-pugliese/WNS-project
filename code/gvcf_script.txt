#!/bin/bash
#SBATCH --job-name=GVCF
#SBATCH -N 1
#SBATCH -n 1
#SBATCH -c 4
#SBATCH --mem=20G
#SBATCH --partition=general
#SBATCH --qos=general
#SBATCH --mail-type=ALL
#SBATCH --mail-user=caprina.pugliese@uconn.edu
#SBATCH -o %x_%j.out
#SBATCH -e %x_%j.err

########### script start

hostname
date

## load modules
module load GATK/4.3.0.0
module load parallel/20180122

## load variables
ACCLIST=/home/FCAM/cpugliese/wns/02_bam2gvcf/03_accessionlist/acclist_without_Pd03.txt
#### point to paths
INDIR=/home/FCAM/cpugliese/wns-lab/bams
OUTDIR=/home/FCAM/cpugliese/wns-lab/gvcfs
REF=/home/FCAM/cpugliese/wns/pd_data/pd_ref/pdestructans.fasta

## files need to be indexed
#### run indexing script first

## Make .gvcf from sorted bam
cd $OUTDIR

cat $ACCLIST | parallel -j 5 \
  gatk HaplotypeCaller \
    -R $REF \
    -I $INDIR/{}.deduped.bam \
    -O {}.g.vcf.gz \
    -ERC GVCF \
    -ploidy 1

echo "make .gvcf done"
date

########### script end
